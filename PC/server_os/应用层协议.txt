#os命令
motor auto       #自动舵机
motor [angele]  #转到特定角度吧
motor manual       #手动舵机
detect      #启动检测程序
monitor     #启动监视器
cam on      #启动相机
cam off       #关闭相机（开启需在相机ssl上执行）
state           #查看工作状态

#tcp str数据格式
#客户端询问格式：
[m/c]  #m舵机，c相机
#客户端传输格式
[t]+[角度编号]   #rtsp模式检测
[t]+[角度编号]+[按顺序的图片数组数据]   #边缘计算分割网模式
[t]+[角度编号]+[1/0]    #边缘计算分类网模式
#服务端回答格式
[m/c]+[1/0]+[1/0]   
#服务端询问格式
[a]
#客户端回答格式
[a]+[舵机状态]+[执行状态]+[相机状态]
#请求异常
[x]
